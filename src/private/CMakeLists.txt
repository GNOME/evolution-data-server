)

	-DG_LOG_DOMAIN=\"libedbus-private\"

	${DATA_SERVER_CFLAGS}
	${GIO_UNIX_CFLAGS}
)

	${CMAKE_BINARY_DIR}
	${CMAKE_BINARY_DIR}/src
	${DATA_SERVER_INCLUDE_DIRS}
	${GIO_UNIX_INCLUDE_DIRS}
)

	${DATA_SERVER_LDFLAGS}
	${GIO_UNIX_LDFLAGS}
)

	DESTINATION ${privsolibdir}
